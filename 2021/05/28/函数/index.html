函数提高

函数默认参数

在C++中，函数的形参列表中的形参可以有默认值。

如果函数中某个位置的参数有默认值，那么这个参数后面的所有参数都要有默认值。

如果函数中的参数有默认值，如果传入了参数就用传入的实参，如果没有传入实参，就用默认值。

如果函数声明有默认参数，函数实现就不能有默认参数。

#include <iostream>
using namespace std;

int func(int a = 10, int b = 20, int c = 30)
{
    return a + b + c;
}
/*int func2(int a = 10, int b, int c)
{
    return a + b + c;
}a之后的参数要有默认参数，当实参的某个位置有默认参数，其后的参数必须有默认参数*/
int func3(int a = 10, int b = 10);
int func3(int a, int b)
{
    return a + b;
}   //函数实现和函数声明中只能有一个有默认参数
int main()
{
    cout << func() << endl;
    cout << func(10, 10, 10) << endl;
    cout << func3() << endl;
    return 0;
}

函数占位参数

#include <iostream>
using namespace std;
void func(int a, int)    //占位参数int
{
    cout << "This is a 函数" << endl;
}
int main()
{
    func(10, 10);    //调用函数时必须把占位参数补上
    return 0;
}

函数重载

函数的名称允许相同，提高复用性。

函数重载的条件：

  • 在同一个作用域下
  • 函数名相同
  • 函数的数据类型相同
  • 函数的参数类型不同或者参数类型个数不同或参数顺序不同
#include <iostream>
#include <cstring>
using namespace std;

void func1(int a)
{
    cout << "a=" << a << endl;
}
void func1(string s)
{
    cout << s << endl;
}
void func1(char c)
{
    cout << "this is a char and the char is " << c << endl;
}
void func1(int a, double b)
{
    cout << "a=" << a << " b=" << b << endl;
}
void func1(double a, int b)
{
    cout << "a=" << a << " b=" << b << endl;
}
int main()
{
    func1(10);
    func1("joker");
    func1('M');
    func1(10, 3.14);
    func1(3.14, 10);
    return 0;
}

函数重载注意事项

#include <iostream>
using namespace std;
// 引用作为重载的条件
void func(int &a)
{
    cout << "int a = " << a << endl;
}
void func(const int &a)
{
    cout << "const int a = " << a << endl;
}

// 函数重载碰到默认参数
void func2(int a)
{
    cout << "func2 int a = " << a << endl;
}

void func2(int a = 10, int b = 10)
{
    cout << "func2 int a = 10 " << a << endl;
}


int main()
{
    int a = 10; //调用func(int &a)，因为a是变量，可读可写，被const修饰的a变为只读的
    func(a);
    func(10); //调用func(const int &a)    无法int &a = 10;
    //func2(10);    当函数重载碰到默认参数时，会出现二义性，尽量避免。
    return 0;
}
