#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
int n, m, flag;
const int N = 5000005;
int head[N], tot;
struct edge
{
    int to, next, w;
} edge[N];
void add_edge(int u, int v, int weight)
{
    edge[++tot].to = v;
    edge[tot].next = head[u];
    edge[tot].w = weight;
    head[u] = tot;
}
void print_edge()
{
    for (int i = 1; i <= n; i++)
    {
        for (int j = head[i]; j != 0; j = edge[j].next)
        {
            cout << i << " " << edge[j].to << " " << edge[j].w;
            cout << endl;
        }
    }
}
int main()
{
    cin >> n >> m >> flag;
    for (int i = 1; i <= m; i++)
    {
        int a, b, weight;
        cin >> a >> b >> weight;
        add_edge(a, b, weight);
        if (flag == 0)
        {
            add_edge(b, a, weight);
        }
    }
    print_edge();
    return 0;
}

最近公共祖先

P3379 【模板】最近公共祖先（LCA）

朴素写法

#include <iostream>
#include <algorithm>
#include <cstdio>
#include <vector>
#include <cstring>
using namespace std;
const int N = 50005;
int n, m, s, a, b;
int vis[N], p[N];
vector<int> v[N];

void dfs(int x)
{
    int mep = v[x].size();
    for (int i = 0; i < mep; i++)
    {
        int u = v[x][i];
        if (u != p[x])
        {
            p[u] = x;
            dfs(u);
        }
    }
}
int lca(int x, int y)
{
    memset(vis, 0, sizeof(vis));
    while (x)
    {
        vis[x] = true;
        x = p[x];
    }
    while (!vis[y])
        y = p[y];
    return y;
}

int main()
{
    cin >> n >> m >> s;
    for (int i = 1; i < n; i++)
    {
        cin >> a >> b;
        v[a].push_back(b);
        v[b].push_back(a);
    }
    dfs(s);
    while (m--)
    {
        cin >> a >> b;
        cout << lca(a, b) << endl;
    }
    return 0;
}

deep写法

#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>
#include <vector>
using namespace std;
const int N = 200020;
int n, m, s, a, b, dep[N], p[N];
vector<int> v[N];
void dfs(int u, int d)
{
    int mp = v[u].size();
    dep[u] = d;
    for (int i = 0; i < mp; i++)
    {
        int x = v[u][i];
        if (x != p[u])
        {
            p[x] = u;
            dfs(x, d + 1);
        }
    }
}
int lca(int x, int y)
{
    if (dep[x] < dep[y])
        swap(x, y);
    while (dep[x] != dep[y])
        x = p[x];
    while (x != y)
    {
        x = p[x];
        y = p[y];
    }
    return x;
}
int main()
{
    cin >> n >> m >> s;
    for (int i = 1; i < n; i++)
    {
        cin >> a >> b;
        v[a].push_back(b);
        v[b].push_back(a);
    }
    dfs(s, 0);
    while (m--)
    {
        cin >> a >> b;
        cout << lca(a, b) << endl;
    }
    return 0;
}

倍增法

#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <vector>
#include <assert.h>
using namespace std;
const int N = 300005;
int n, m, s, a, b, f[N][23], deep[N];
vector<int> v[N];
void dfs(int u, int p, int d)
{
    f[u][0] = p;
    deep[u] = d;
    int sm = v[u].size();
    for (int i = 0; i < sm; i++)
    {
        int next = v[u][i];
        if (next != p)
        {
            dfs(next, u, d + 1);
        }
    }
}
int lca(int x, int y)
{
    if (deep[x] < deep[y])
        swap(x, y);
    for (int i = log2(deep[x] - deep[y]); i >= 0; i--)
    {
        if ((1 << i) <= (deep[x] - deep[y]))
            x = f[x][i];
    }
    if (x == y)
        return x;
    for (int i = log2(deep[x]); i >= 0; i--)
    {
        if (f[x][i] != f[y][i])
        {
            x = f[x][i];
            y = f[y][i];
        }
    }
    return f[x][0];
}
int main()
{
    cin >> n >> m >> s;
    for (int i = 1; i < n; i++)
    {
        cin >> a >> b;
        v[a].push_back(b);
        v[b].push_back(a);
    }
    dfs(s, s, 0);
    for (int i = 1; i < 23; i++)
    {
        for (int j = 1; j <= n; j++)
        {
            f[j][i] = f[f[j][i - 1]][i - 1];
        }
    }
    while (m--)
    {
        cin >> a >> b;
        cout << lca(a, b) << endl;
    }
    return 0;
}

#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <vector>
using namespace std;
const int N = 200050;
int n, m, a, b, tot, f[N][25], deep[N], root[N], head[N], ans[N];
int maxAns;
vector<int> v[N];
struct edge
{
    int to, next;
    /* data */
} edge[N];
void add_edge(int x, int y)
{
    edge[++tot].next = head[x];
    edge[tot].to = y;
    head[x] = tot;
}
void dfs(int u, int vt, int d)
{
    f[u][0] = vt;
    deep[u] = d;
    for (int i = 0; i < v[u].size(); i++)
    {
        int next = v[u][i];
        if (next != vt)
        {
            dfs(next, u, d + 1);
        }
    }
}
int lca(int x, int y)
{
    if (deep[x] < deep[y])
    {
        swap(x, y);
    }
    for (int i = log2(deep[x] - deep[y]); i >= 0; i--)
    {
        if ((1 << i) <= (deep[x] - deep[y]))
        {
            x = f[x][i];
        }
    }
    if (x == y)
        return x;
    for (int i = log2(deep[x]); i >= 0; i--)
    {
        if (f[x][i] != f[y][i])
        {
            x = f[x][i];
            y = f[y][i];
        }
    }
    return f[x][0];
}
void insert(int u, int v)
{
    ans[u]++, ans[v]++;
    int Lca = lca(u, v);
    ans[Lca]--;
    ans[f[Lca][0]]--;
}
void sum(int u)
{
    for (int i = head[u]; i != 0; i = edge[i].next)
    {
        int v = edge[i].to;
        if (v != f[u][0])
        {
            sum(v);
            ans[u] += ans[v];
        }
    }
    maxAns = max(maxAns, ans[u]);
}
int main()
{
    cin >> n >> m;
    for (int i = 1; i < n; i++)
    {
        cin >> a >> b;
        v[a].push_back(b);
        v[b].push_back(a);
        add_edge(a, b);
        add_edge(b, a);
    }
    dfs(1, 1, 0);
    for (int i = 1; i < 22; i++)
    {
        for (int j = 1; j <= n; j++)
        {
            f[j][i] = f[f[j][i - 1]][i - 1];
        }
    }
    for (int i = 1; i <= m; i++)
    {
        cin >> a >> b;
        insert(a, b);
    }
    sum(1);
    cout << maxAns;
    return 0;
}

tarjan算法

关于图的一些连通性的定义

  • 连通

    无向图中，从任意节点i可达到任意节点j

  • 强连通

    有向图中，从任意节点i可达到任意节点j，每个结点可看作强连通

  • 弱连通

    把有向图看作无向图时，从任意节点i可达到任意节点j

  • 强连通分量

    有向图的极大强连通子图，称为强连通分量

tarjan算法求强连通分量

dfn[N]:结点的编号。
p[N]:结点的时间戳，即结点能够通过回路返回到最小的结点时间戳，如果多个结点的时间戳相同时，
就表示这几个点构成了一个强连通。
如图：

其中(i,j)中的i表示结点的编号，j表示该点的时间戳，即p[j]，可以看到b、c、d构成了一个回路，a、f、g也构成了一个回路，b为c和d能够通过有向边回到的最小的结点，所以d的时间戳变为b的时间戳，c能够通过d回到b，所以c的时间戳也变为b的时间戳，a、f、g同理，然后再寻找构成的回路中所包含的最多结点的回路，这样就找到了强连通分量。

  • 问题
    输入两个整数n和m，n代表图的结点数(从1到n)，接下来m行，每行两个整数u、v表示有向图的一条边又u->v,求图中能构成强连同分量的结点。
  • 输入样例

    7 9
    1 2
    2 3
    3 4
    3 5
    4 5
    1 6
    6 7
    7 1
    4 2

  • 输出样例

    1 6 7
    2 3 4

    #include <iostream>
#include <algorithm>
#include <stack>
#include <vector>
using namespace std;
const int N = 100010;
vector<int> v[N];
vector<int> ans[N];
int maxx;
stack<int> s;
int n, m, a, b, dfn[N], p[N], cur = 1; //dfn当前结点的编号，p当前结点的时间戳
                                     //如果多个结点时间戳相同，则这多个结点构成强连通
typedef struct node
{
  int nodeNum, nodeWay;
} Node;
Node NodeM[N];
int cmp(Node x, Node y)
{
  return x.nodeNum > y.nodeNum;
}
bool search(int x)
{
  int t = s.size();
  int countS[N];
  bool flag = false;
  int i = t;
  while (!s.empty())
  {
      if (x == s.top())
      {
          flag = true;
      }
      countS[i--] = s.top();
      s.pop();
  }
  for (i = 1; i <= t; i++)
  {
      s.push(countS[i]);
  }
  return flag;
}
void dfs(int x)
{
  dfn[x] = cur;
  p[x] = cur++;
  s.push(x);
  for (int i = 0; i < v[x].size(); i++)
  {
      if (v[x][i]) //点x是否有出度
      {
          if (dfn[v[x][i]] == 0) //点x所指向的点是否搜过，没搜过就dfs搜索
          {
              dfs(v[x][i]);
              p[x] = min(p[x], p[v[x][i]]);
          }
          else if (search(v[x][i])) //如果搜过了，判断它是否还在栈中
          {
              p[x] = min(p[x], p[v[x][i]]); //如果在栈中，就取时间戳最小的
          }
      }
  }
  if (dfn[x] == p[x]) //如果结点标号等于时间戳，表明该点没有出度
  {
      s.pop();
  }
}
int main()
{
  cin >> n >> m;
  int temp = 0;
  for (int i = 1; i <= m; i++)
  {
      cin >> a >> b;
      v[a].push_back(b);
  }
  for (int i = 1; i <= n; i++)
  {
      if (dfn[i] == 0) //如果该点的编号为0，表示该点还没有被搜过，就进行搜索
      {
          dfs(i);
      }
  }
  for (int i = 1; i <= n; i++)
  {
      ans[p[i]].push_back(i);
  }
  for (int i = 1; i <= n; i++)
  {
      /*if (ans[i].size() > maxx)
      {
          maxx = ans[i].size();
          temp = i;
      }*/
      NodeM[i].nodeNum = ans[i].size();
      NodeM[i].nodeWay = i;
  }
  maxx = 0;
  sort(NodeM + 1, NodeM + 1 + n, cmp);
  /*for (int i = 1; i <= n; i++)
  {
      cout << NodeM[i].nodeNum << NodeM[i].nodeWay << endl;
  }*/
  for (int i = 1; i <= n; i++)
  {
      //cout << maxx << endl;
      if (maxx > NodeM[i].nodeNum)
      {
          break;
      }
      maxx = NodeM[i].nodeNum;
      for (int j = 0; j < maxx; j++)
      {
          cout << ans[NodeM[i].nodeWay][j] << " ";
      }
      cout << endl;
  }
  /*for (int i = 0; i < maxx; i++)
  {
      cout << ans[temp][i] << " ";
  }
  cout << endl;*/
  return 0;
}

dijkstra算法求最短路

先从点1出发，找与点1相连的距离最短的点的，并把这个点标为已搜过，然后再从这个点出发，找与该点相连的点的距离最近并没有搜过的点，然后重复操作，直到找到从点1出发到各个顶点的最短距离，如果有点无法到达，则距离为正无穷。

  • 参考代码
    #include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
#define inf 0x3f3f3f3f
const int N = 100010;
int n, m;
int map[2000][2000];
int dis[N], vis[N];
void dijkstra(int u)
{
  memset(vis, 0, sizeof(vis));
  for (int t = 1; t <= n; t++)
  {
      dis[t] = map[u][t];
  }
  vis[u] = 1;
  dis[u] = 0;
  for (int t = 1; t <= n; t++)
  {
      int minn = inf, temp;
      for (int i = 1; i <= n; i++)
      {
          if (!vis[i] && dis[i] < minn)
          {
              minn = dis[i];
              temp = i;
          }
      }
      vis[temp] = 1;
      for (int i = 1; i <= n; i++)
      {
          if (dis[i] > map[temp][i] + dis[temp])
          {
              dis[i] = map[temp][i] + dis[temp];
          }
      }
  }
}
int main()
{
  cin >> n >> m;
  memset(map, inf, sizeof(map));
  for (int i = 1; i <= m; i++)
  {
      int u, v, w;
      cin >> u >> v >> w;
      if (map[u][v] > w)
      {
          map[u][v] = w;
      }
  }
  dijkstra(1);
  for (int i = 1; i <= n; i++)
  {
      cout << dis[i] << " ";
  }
  return 0;
}
