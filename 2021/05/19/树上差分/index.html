链式前向星

#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
int n, m, flag;
const int N = 5000005;
int head[N], tot;
struct edge
{
    int to, next, w;
} edge[N];
void add_edge(int u, int v, int weight)
{
    edge[++tot].to = v;
    edge[tot].next = head[u];
    edge[tot].w = weight;
    head[u] = tot;
}
void print_edge()
{
    for (int i = 1; i <= n; i++)
    {
        for (int j = head[i]; j != 0; j = edge[j].next)
        {
            cout << i << " " << edge[j].to << " " << edge[j].w;
            cout << endl;
        }
    }
}
int main()
{
    cin >> n >> m >> flag;
    for (int i = 1; i <= m; i++)
    {
        int a, b, weight;
        cin >> a >> b >> weight;
        add_edge(a, b, weight);
        if (flag == 0)
        {
            add_edge(b, a, weight);
        }
    }
    print_edge();
    return 0;
}

求路径上的点的权值

P3128 [USACO15DEC]Max Flow P

#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <vector>
using namespace std;
const int N = 200050;
int n, m, a, b, tot, f[N][25], deep[N], root[N], head[N], ans[N];
int maxAns;
vector<int> v[N];
struct edge
{
    int to, next;
    /* data */
} edge[N];
void add_edge(int x, int y)
{
    edge[++tot].next = head[x];
    edge[tot].to = y;
    head[x] = tot;
}
void dfs(int u, int vt, int d)
{
    f[u][0] = vt;
    deep[u] = d;
    for (int i = 0; i < v[u].size(); i++)
    {
        int next = v[u][i];
        if (next != vt)
        {
            dfs(next, u, d + 1);
        }
    }
}
int lca(int x, int y)
{
    if (deep[x] < deep[y])
    {
        swap(x, y);
    }
    for (int i = log2(deep[x] - deep[y]); i >= 0; i--)
    {
        if ((1 << i) <= (deep[x] - deep[y]))
        {
            x = f[x][i];
        }
    }
    if (x == y)
        return x;
    for (int i = log2(deep[x]); i >= 0; i--)
    {
        if (f[x][i] != f[y][i])
        {
            x = f[x][i];
            y = f[y][i];
        }
    }
    return f[x][0];
}
void insert(int u, int v)
{
    ans[u]++, ans[v]++;
    int Lca = lca(u, v);
    ans[Lca]--;
    ans[f[Lca][0]]--;
}
void sum(int u)
{
    for (int i = head[u]; i != 0; i = edge[i].next)
    {
        int v = edge[i].to;
        if (v != f[u][0])
        {
            sum(v);
            ans[u] += ans[v];
        }
    }
    maxAns = max(maxAns, ans[u]);
}
int main()
{
    cin >> n >> m;
    for (int i = 1; i < n; i++)
    {
        cin >> a >> b;
        v[a].push_back(b);
        v[b].push_back(a);
        add_edge(a, b);
        add_edge(b, a);
    }
    dfs(1, 1, 0);
    for (int i = 1; i < 22; i++)
    {
        for (int j = 1; j <= n; j++)
        {
            f[j][i] = f[f[j][i - 1]][i - 1];
        }
    }
    for (int i = 1; i <= m; i++)
    {
        cin >> a >> b;
        insert(a, b);
    }
    sum(1);
    cout << maxAns;
    return 0;
}
