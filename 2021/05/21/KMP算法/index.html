KMP算法

2021-05-21- 2021-05-24
算法-必须掌握
  • 字符串匹配的高效算法
    通过next数组记录模式串的子串的最长相同前后缀的长度。

    等到next数组后，模式串与匹配串进行匹配，如果遇到不相等的字符，就返回当前模式串的前一个字符的所对应的next数组的值，直到返回到模式串的首位或返回到的字符等于匹配串当前所搜索的字符时，停止前跳。如果相同，就令模式串的下标+1。然后更新next数组。

  • 参考代码
    #include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
const int N = 100010;
int nextt[N], ans, n;
string s, str;
void getNext()
{
  nextt[0] = 0;
  int j = 0;
  for (int i = 1; i < s.size(); i++)
  {
      while (j > 0 && s[j] != s[i])
      {                     //前后缀不相同
          j = nextt[j - 1]; // 向前找
      }
      if (s[i] == s[j])
      {
          j++;
      }
      nextt[i] = j;
  }
}
void getAns()
{
  int j = 0;
  for (int i = 0; i < str.size(); i++)
  {
      while (j > 0 && s[j] != str[i])
      {
          j = nextt[j - 1];
          /* code */
      }
      if (s[j] == str[i])
      {
          j++;
      }
      if (j == s.size())
      {
          ans++;
          i = i - j + 1;
          j = 0;
      }
  }
}
bool getBool(string &Str)
{
  for (int i = 0; i < s.size(); i++)
  {
      if (s[i] != Str[i])
      {
          return false;
      }
  }
  return true;
}
int main()
{
  cin >> n;
  while (n--)
  {
      ans = 0;
      cin >> s >> str;
      getNext();
      getAns();
      string sp = s;
      reverse(s.begin(), s.end());
      if (getBool(sp))
      {
          cout << ans << endl;
          continue;
      }
      memset(nextt, 0, sizeof(nextt));
      getNext();
      getAns();
      cout << ans << endl;
  }
  return 0;
}
