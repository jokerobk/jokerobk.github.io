立方体

判断两个立方体是否相等，并计算立方体的面积和体积

  • 引用方式
#include <iostream>
using namespace std;
class cube
{
private:
    int m_H;
    int m_W;
    int m_L;

public:
    void setInfo(int m_H, int m_W, int m_L);
    int getm_H();
    int getm_W();
    int getm_L();
    int getMianji();
    int getV();
    bool isxiangdeng(cube &c);
};
void cube::setInfo(int m_H, int m_W, int m_L)
{
    this->m_H = m_H;
    this->m_W = m_W;
    this->m_L = m_L;
}
int cube::getm_H()
{
    return m_H;
}
int cube::getm_W()
{
    return m_W;
}
int cube::getm_L()
{
    return m_L;
}
int cube::getMianji()
{
    return 2 * (m_H * m_L + m_H * m_W + m_L * m_W);
}
int cube::getV()
{
    return m_H * m_W * m_L;
}
bool cube::isxiangdeng(cube &c)
{
    if (m_H == c.m_H && m_L == c.m_L && m_W == c.m_W)
    {
        return true;
    }
    else
    {
        return false;
    }
}

int main()
{
    cube c1;
    cube c2;
    c1.setInfo(10, 10, 10);
    c2.setInfo(10, 50, 10);
    if (c1.getm_H() == c2.getm_H() && c1.getm_L() == c2.getm_L() && c1.getm_W() == c2.getm_W())
    {
        cout << "Y" << endl;
    }
    else
    {
        cout << "N" << endl;
    }
    int flag = c1.isxiangdeng(c2);
    if (flag)
    {
        cout << "Y" << endl;
    }
    else
    {
        cout << "N" << endl;
    }
    cout << "c1的面积: " << c1.getMianji() << "c1的体积: " << c1.getV() << endl;
    cout << "c2的面积: " << c2.getMianji() << "c2的体积: " << c2.getV() << endl;
    return 0;
}

判断点和圆的关系

#include <iostream>
#include <cmath>
using namespace std;
class point
{
private:
    double px, py;

public:
    void setInfo(double px, double py);
    double getPx()
    {
        return px;
    }
    double getPy()
    {
        return py;
    }
};
class cir
{
private:
    double cx, cy;
    double cr;

public:
    void setInfo(double cx, double cy, double cr);
    double getCx()
    {
        return cx;
    }
    double getCy()
    {
        return cy;
    }
    double getCr()
    {
        return cr;
    }
};
class guanxi
{
private:
    double length;

public:
    void len(point &p, cir &c);
    void isPansC(cir &c);
    void setInfo();
    double getLength()
    {
        return length;
    }
};
void point::setInfo(double px, double py)
{
    this->px = px;
    this->py = py;
}
void cir::setInfo(double cx, double cy, double cr)
{
    this->cx = cx;
    this->cy = cy;
    this->cr = cr;
}
void guanxi::len(point &p, cir &c)
{
    length = sqrt(pow(abs(p.getPx() - c.getCx()), 2) + pow(abs(p.getPy() - c.getCy()), 2));
}

void guanxi::isPansC(cir &c)
{
    if (length > c.getCr())
    {
        cout << "点在圆外" << endl;
    }
    else if (length == c.getCr())
    {
        cout << "点在圆上" << endl;
    }
    else
    {
        cout << "点在圆内" << endl;
    }
}

int main()
{
    point p;
    cir c;
    guanxi g;
    p.setInfo(0, 0);
    c.setInfo(0, 0, 0);
    g.len(p, c);
    g.isPansC(c);
    return 0;
}
