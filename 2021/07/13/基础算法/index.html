1、快速排序 （ 快速排序）

解析：
1.首先设立一个X将数组分开成两个数组；
2.设立两个指针i和j分别从首尾两端向彼此靠近，将小于X的数据都放在X的左边，大于X的数据放在X的右边。当两个指针相遇时，继续下一步；
3.将左边的数据和右边的数据继续进行上面的操作；
4.重复上述过程，可以看出这是一个递归的过程；当左右两边的数据都排好序之后，整个数组就自然有序了。

#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
void quick_sort(int a[], int l, int r)
{
    if (l >= r)
        return;
    int i = l - 1, j = r + 1, x = a[(l + r >> 1)];
    while (i < j)
    {
        do
            i++;
        while (a[i] < x);
        do
            j--;
        while (a[j] > x);
        if (i < j)
            swap(a[i], a[j]);
    }
    quick_sort(a, l, j);
    quick_sort(a, j + 1, r);
}
int main()
{
    int n;
    cin >> n;
    int a[10010];
    for (int i = 1; i <= n; i++)
    {
        cin >> a[i];
    }
    quick_sort(a, 1, n);
    for (int i = 1; i <= n; i++)
    {
        cout << a[i] << " ";
    }
    return 0;
}

2、归并排序 （归并排序）

#include <iostream>
using namespace std;
void memr(int a[], int l, int mid, int r, int temp[])
{
    int i = l, j = mid + 1;
    int z = 0;
    while (i <= mid && j <= r)
    {
        if (a[i] > a[j])
        {
            temp[z++] = a[j++];
        }
        else
        {
            temp[z++] = a[i++];
        }
    }
    while (i <= mid)
    {
        temp[z++] = a[i++];
    }
    while (j <= r)
    {
        temp[z++] = a[j++];
    }
    z = 0;
    int leftz = l;
    while (leftz <= r)
    {
        a[leftz++] = temp[z++];
    }
}
void memrge(int a[], int l, int r, int temp[])
{
    if (l < r)
    {
        int mid = (l + r) >> 1;
        memrge(a, l, mid, temp);
        memrge(a, mid + 1, r, temp);
        memr(a, l, mid, r, temp);
    }
}
int main()
{
    int n, a[100010], temp[100010];
    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        cin >> a[i];
    }
    memrge(a, 1, n, temp);
    for(int i = 1; i <= n; i ++){
        cout << a[i] << " ";
    }
    return 0;
}

3、整数二分法算法模版 （数的范围）

#include <iostream>
using namespace std;
int n, q, k, a[100010];
int main()
{
    cin >> n >> q;
    for (int i = 1; i <= n; i++)
    {
        cin >> a[i];
    }
    while (q--)
    {
        cin >> k;
        int l = 1, r = n;
        while (l < r)
        {
            int mid = (l + r) >> 1;
            if (a[mid] >= k)
                r = mid;
            else
                l = mid + 1;
        }
        // # 找到想要查询数字的最小的数组下标
        if (a[l] != k)
        {
            cout << -1 << " " << -1 << endl;
        }
        else
        {
            cout << l - 1 << " ";
            l = 1, r = n;
            while (l < r)
            {
                int mid = (l + r + 1) >> 1;
                if (a[mid] > k)
                    r = mid - 1;
                else
                    l = mid;
            }
            // 找到想要查询数字的最大的数组下标
            cout << l - 1 << endl;
        }
    }
    return 0;
}

4、浮点数二分法 （数的三次方根）

#include <iostream>
#include <iomanip>
using namespace std;
double n;
double ans(double a)
{
    return a * a * a;
}
double erfen(double l, double r)
{
    while (r - l >= 1e-7)
    {
        double mid = (l + r) / 2;
        // cout << l << " " << r << endl;
        // cout << mid << endl;
        if (ans(mid) >= n)
        {
            r = mid;
        }
        else
        {
            l = mid;
        }
    }
    return l;
}
int main()
{
    cin >> n;
    double l = -10000;
    double r = 10000;
    cout << fixed << setprecision(6) << erfen(l, r);
    // printf("%d", 1e-7);
    return 0;
}

5、一维差分 （差分）

#include<iostream>
using namespace std;
int n, m, l, r, c;
int a[100010], b[100010];
int main(){
    cin >> n >> m;
    for(int i = 1; i <= n; i++){
        cin >> a[i];
        b[i] = a[i] - a[i - 1];
    }
    while(m--){
        cin >> l >> r >> c;
        b[l] += c;
        b[r + 1] -= c;
    }
    for(int i = 1; i <= n; i ++){
        b[i] += b[i - 1];
        cout << b[i] << " ";
    }
    return 0;
}
