#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
const int N = 1e6 + 10;
typedef long long ll;
int n, q;
ll a[N], t[N];
int lowbit(int x){
    return x & -x;
}
void add(int x, int k){
    for(int i = x; i <= n; i += lowbit(i)){
        t[i] += k;
    }
}
ll sum(int x){
    ll res = 0;
    for(int i = x; i; i -= lowbit(i)){
        res += t[i];
    }
    return res;
}
int main(){
    cin >> n >> q;
    for(int i = 1; i <= n; i ++){
        cin >> a[i];
        add(i, a[i]);
    }
    while(q --){
        int x, a, b;
        cin >> x >> a >> b;
        if(x == 1){
            add(a, b);
        }
        else{
            cout << sum(b) - sum(a - 1) << endl;
        }
    }
    return 0;
}

#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
typedef long long ll;
const int N = 1e6 + 10;
int n, q;
ll a[N], b[N], t1[N], t2[N];
int lowbit(int x){
    return x & -x;
}
void add(int x, ll k){
    ll v = x * k;
    for(int i = x; i <= n; i += lowbit(i)){
        t1[i] += k;
        t2[i] += v;
    }
}
ll sum(int x, ll *t){
    ll res = 0;
    for(int i = x; i; i -= lowbit(i)){
        res += t[i];
    }
    return res;
}
ll getSum(ll l, ll r){
    return (r + 1ll) *  sum(r, t1) - l * sum(l - 1, t1) - (sum(r, t2) - sum(l - 1, t2));
}

int main(){
    cin >> n >> q;
    for(int i = 1; i <= n; i ++){
        cin >> a[i];
        b[i] = a[i] - a[i - 1];
        add(i, b[i]);
    }
    while(q --){
        int flag;
        cin >> flag;
        if(flag == 1){
            int l, r, x;
            cin >> l >> r >> x;
            add(l, x);
            add(r + 1, -x);
        }
        else{
            ll l, r;
            cin >> l >> r;
            cout << getSum(l, r) << endl;
        }
    }
    return 0;
}
